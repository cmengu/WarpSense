  centerTemp: number,
  variation: number
): ThermalSnapshot {
  const data: number[][] = [];
  
  // 48x48 grid
  for (let y = 0; y < 48; y++) {
    const row: number[] = [];
    for (let x = 0; x < 48; x++) {
      // Distance from center
      const dx = x - 24;
      const dy = y - 24;
      const distance = Math.sqrt(dx * dx + dy * dy);
      
      // Temperature decreases with distance from center
      const temp = centerTemp - (distance * 10) + (Math.random() - 0.5) * variation;
      row.push(Math.max(400, Math.min(1600, temp))); // Clamp to realistic range
    }
    data.push(row);
  }
  
  return {

  };

# Task:

 Migration is the highest-risk step and the pre-flight is insufficient
Step 1.2 is the only irreversible step in the plan and the pre-flight for it is just "check alembic history." The single UPDATE sessions SET process_type = 'mig' with a subsequent ALTER COLUMN ... nullable=False is a two-step operation that can leave the table in a broken state if anything fails between them. If the UPDATE succeeds but the ALTER fails (connection drop, timeout), you have a nullable column with data — and the next alembic upgrade attempt will fail because the revision is partially applied.
The Known Issues section documents the large-table concern but the mitigation is "run during maintenance window." For a dev environment with mock data this is fine. But the pre-flight checklist doesn't distinguish dev from staging/prod at all, which means someone running this on a real DB follows the same checklist as someone on a fresh local DB.
Fix: Add to Step 1.2 pre-flight: "If sessions table has >1000 rows, run the UPDATE manual first and verify row count before running alembic upgrade." Also add alembic upgrade head --sql > migration_preview.sql as a pre-flight step so the operator can inspect what will run.

Problem 2: The cache has a correctness bug under concurrent requests
_load_all uses a double-check locking pattern with _load_lock, but the outer check if not _cache_loaded in get_thresholds is outside the lock. Two concurrent requests can both see _cache_loaded = False, both enter _load_all, the first acquires the lock and loads, sets _cache_loaded = True, releases. The second acquires the lock, sees _cache_loaded = True (double-check passes), returns. This is actually fine for loading.
The real bug is invalidate_cache: it sets _cache_loaded = False without the lock. A request mid-way through _load_all (has acquired lock, is building _threshold_cache) can have _cache_loaded = False set by invalidate_cache after the lock check but before _cache_loaded = True is written. The next request then sees _cache_loaded = False and calls _load_all again, which is harmless but means the invalidate_cache in finally of the PUT handler runs before the new DB values are committed (since finally runs after db.commit()). Wait — actually finally runs after db.commit() succeeds, so the cache is cleared after the write. That part is fine. But invalidate_cache should also acquire _load_lock to avoid the race.
Fix: Have invalidate_cache acquire _load_lock before setting _cache_loaded = False, or use a single lock for both read and write operations. Document the threading model explicitly in the service docstring.

Problem 3: Step 3.4 form has a silent NaN bug that the verification misses
The form uses parseFloat(e.target.value) || 0 for all numeric inputs. This means if a user types 0 for any field (e.g. amps_stability_warning = 0), the expression parseFloat("0") || 0 evaluates to 0 || 0 = 0 — which is correct. But if a user types 0.5, parseFloat("0.5") || 0 = 0.5 — also correct. The real problem is parseFloat("") || 0 = 0, which means clearing a field sets it to 0 silently rather than an invalid state. isCompleteForm checks Number.isFinite(v) but 0 is finite, so a cleared field passes validation and submits 0 to the backend.
For most fields 0 is technically valid (Pydantic allows ge=0). But angle_target_degrees=0 with angle_warning_=0 means every weld angle >0 triggers a warning — the scoring becomes useless. The backend doesn't reject it because 0 passes ge=0.
Fix: Change parseFloat(e.target.value) || 0 to parseFloat(e.target.value) (no fallback), and update isCompleteForm to check typeof v === 'number' && Number.isFinite(v) && v >= 0 but also add a specific check: angle_target_degrees > 0. Or add backend validation rejecting angle_target_degrees === 0.

Problem 4: Step 4.6 test fixture is underspecified and likely to fail
test_put_threshold_invalidates_cache depends on a session_factory fixture that creates "a minimal session with frames + process_type." This fixture isn't defined anywhere in the plan. Creating a valid session with frames in an in-memory SQLite test requires the full Session model, FrameModel, and all related tables — the same complexity that makes the mock_welders plan's test fixtures non-trivial.
The plays "Fixtures: db (in-memory SQLite, Base.metadata.create_all — ensure WeldThresholdModel imported first)" but doesn't explain what session_factory produces or how frames are generated. If the implementer uses a minimal session without frames, get_session_score may fail before hitting the threshold logic, and the test passes for the wrong reason (404 on session not found).
Fix: Provide explicit session_factory fixture code in Step 4.6, similar to how Step 5.1 in the mock welders plan provides full test code. At minimum: "session_factory creates a SessionModel with 10 valid frames using generate_session_for_welder('mike-chen', 'fast_learner', 0, 'sess_test_001') and saves it to the test DB."

Problem 5: Phase 4 micro-feedback wiring has a load order dependency with no guard
Step 4.2 builds thresholdsForMicroFeedback from primaryScore?.active_threshold_spec and passes it to generateMicroFeedback. The useMemo depends on primaryScore?.active_threshold_spec. But on first render, primaryScore is null (score fetch is async). generateMicroFeedback is called with undefined thresholds and falls back to constants — which is the correehavior. On score load, thresholdsForMicroFeedback updates and generateMicroFeedback re-runs with TIG thresholds.
The problem is that sessionData?.frames and primaryScore load independently. If frames load fast and score loads slow, micro-feedback runs twice: once with default thresholds, once with TIG thresholds. This is a visual flicker — micro-feedback markers jump on the replay timeline when the score arrives. For a demo this is visible and looks like a bug.
Fix: Add to Step 4.2 verification: "Verify micro-feedback doesn't re-render visibly when score loads after frames." Mitigation: wait for both sessionData and primaryScore before computing micro-fdback, or show a skeleton on the timeline until both are ready.

**Step 1.4 — Create threshold_service with cache**

```python
"""
Threshold service — cached access to weld_thresholds.
Cache invalidated on PUT; no DB hit per score request.

LIMITATION: In-memory cache is process-local. Multi-worker deployments
(e.g. Gunicorn with multiple uvicorn workers) will serve stale thresholds
until process restart. For MVP: document and add shared cache (e.g. Redis)
to backlog. Single-worker dev/staging: works correctly.
"""

import threading
from typing import Dict, List, Optional

from sqlalchemy import select
from sqlalchemy.orm import Session as OrmSession

from database.models import WeldThresholdModel
from models.thresholds import WeldTypeThresholds

# Module-level cache: weld_type -> WeldpeThresholds
_threshold_cache: Dict[str, WeldTypeThresholds] = {}
_cache_loaded = False
_load_lock = threading.Lock()  # Prevents concurrent _load_all race


def _load_all(db: OrmSession) -> None:
    global _threshold_cache, _cache_loaded
    with _load_lock:
        if _cache_loaded:  # Double-check after acquiring lock
            return
        rows = db.execute(select(WeldThresholdModel)).scalars().all()
        if rows:
            _threshold_cache = {
                r.weld_type: WeldTypeThresholds(
                    weld_type=r.weld_type,
                    angle_target_degrees=r.angle_target_degrees,
                    angle_warning_margin=r.angle_warning_margin,
                    angle_critical_margin=r.angle_critical_margin,
                    thermal_symmetry_warning_celsius=r.thermal_symmetry_warning_celsius,
                    thermal_symmetry_critical_celsius=r.thermal_symmetry_critical_celsius,
                    amps_stability_warning=r.amps_stability_warning,
                    volts_stability_warning=r.volts_stability_warning,
                    heat_diss_consistency=r.heat_diss_consistency,
                )
                for r in rows
            }
        else:
            # Empty table: use hardcoded MIG defaults so scoring doesn't 500
            _threshold_cache = {
                "mig": WeldTypeThresholds(
                    weld_type="mig",
                    angle_target_degrees=45,
                    angle_warning_margin=5,
                    angle_critical_margin=15,
                    thermal_symmetry_warning_celsius=60,
                    thermal_symmetry_critical_celsius=80,
                    amps_stability_warning=5,
                    volts_stability_warning=1,
                    heat_diss_consistency=40,
                )
            }
        _cache_loaded = True


def invalidate_cache() -> None:
    global _cache_loaded
    _cache_loaded = False


def get_thresholds(db: OrmSession, process_type: str) -> WeldTypeThresholds:
    """Get thresholds for a process type. Falls back to mig if unknown."""
    global _cache_loaded
    if not _cache_loaded:
        _load_all(db)
    key = (process_type or "mig").lower()
    if key not in _threshold_cache:
        key = "mig"
    # Guard: _threshold_cache always has "mig" after _load_all (either from DB or fallback)
    return _threshold_cache.get(key, _threshold_cache["mig"])


def get_all_thresholds(db: OrmSession) -> List[WeldTypeThresholds]:
    """Return all thresholds. Admin UI uses this for GET /api/thresholds."""
    global _cache_loaded
    if not _cache_loaded:
        _load_all(db)
    return list(_threshold_cache.values())
```

```python
"""
Thresholds API — GET all, PUT one.

from fastapi import APIRouter, Depends, HTTPException

from database.connection import SessionLocal
from database.models import WeldThresholdModel
from models.thresholds import WeldTypeThresholds, WeldThresholdUpdate
from services.thold_service import invalidate_cache, get_all_thresholds, get_thresholds

router = APIRouter()


def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()


@router.get("/thresholds")
async def list_thresholds(db=Depends(get_db)):
    """Return all thresholds. Admin uses this to populate forms."""
    items = get_all_thresholds(db)
    return [t.model_dump() for t in items]


@router.put("/thresholds/{weld_type}")
async def update_threshold(
    weld_type: str,
    body: WeldThresholdUpdate,
    db=Depends(get_db),
):
    """Update thresholds for one process type. Invalidates cache."""
    weld_type = weld_type.lower()
    if weld_type not in ("mig", "tig", "stick", "flux_core"):
        raise HTTPException(status_code=422, detail=f"Unknown weld_type: {weld_type}")
    if body.angle_warning_margin > body.angle_critical_margin:
        raise HTTPException(
            status_code=422,
            detail="angle_warning_margin must be <= angle_critical_margin",
        )
    if body.thermal_symmetry_warning_celsius > body.thermal_symmetry_critical_celsius:
        raise HTTPException(
            status_code=422,
            detail="thermal_symmetry_warning must be <= thermal_symmetry_critical",
        )
    row = db.query(WeldThresholdModel).filter_by(weld_type=weld_type).first()
    if not row:
        raise HTTPException(status_code=404, detail=f"Thresholds for {weld_type} not found")
    row.angle_target_degrees = body.angle_target_degrees
    row.angle_warning_margin = body.angle_warning_margin
    row.angle_critical_margin = body.angle_critical_margin
    row.thermal_symmetry_warning_celsius = body.thermal_symmetry_warning_celsius
    row.thermal_symmetry_critical_celsius = body.thermal_symmetry_critical_celsius
    row.amps_stability_warning = body.amps_stability_warning
    row.volts_stability_warning = body.volts_stability_warning
    row.heat_diss_consistency = body.heat_diss_consistency
    try:
        db.commit()
    finally:
        # Always invalidate so next request refetches; prevents stale cache if return fails
        invalidate_cache()
    # Build response from committed row — avoids 500 if get_thresholds fails (e.g. connection pool timeout)
    updated = WeldTypeThresholds(
        weld_type=row.weld_type,
        angle_target_degrees=row.angle_target_degrees,
        angle_warning_margin=row.angle_warning_margin,
        angle_critical_margin=row.angle_critical_margin,
        thermal_symmetry_warning_celsius=row.thermal_symmetry_warning_celsius,
        thermal_symmetry_critical_celsius=row.thermal_symmetry_critical_celsius,
        amps_stability_warning=row.amps_stability_warning,
        volts_stability_warning=row.volts_stability_warning,
        heat_diss_consistency=row.heat_diss_consistency,
    )
    return updated.model_dump()
```

```python
from typing import Any, Dict, Optional
from models.scoring import ScoreRule, SessionScore
from models.session import Session
from models.thresholds import WeldTypeThresholds

# Keep as fallback for tests/callers that don't pass thresholds
AMPS_STABILITY_THRESHOLD = 5.0
ANGLE_CONSISTENCY_THRESHOLD = 5.0
THERMAL_SYMMETRY_THRESHOLD = 60.0
HEAT_DISS_CONSISTENCY_THRESHOLD = 40.0
VOLTS_STABILITY_THRESHOLD = 1.0


def score_session(
    session: Session,
    features: Dict[str, Any],
    thresholds: Optional[WeldTypeThresholds] = None,
) -> SessionScore:
    t = thresholds
    rules = [
        _check_amps_stability(features, t),
        _check_angle_consistency(features, t),
        _check_thermal_symmetry(features, t),
        _check_heat_diss_consistency(features, t),
        _check_volts_stability(features, t),
    ]
    passed_count = sum(1 for r in rules if r.passed)
    total = passed_count * 20
    return SessionScore(total=total, rules=rules)


def _check_amps_stability(features: Dict[str, Any], t: Optional[WeldTypeThresholds]) -> ScoreRule:
    actual = features.get("amps_stddev", 0.0)
    th = t.amps_stability_warning if t else AMPS_STABILITY_THRESHOLD
    return ScoreRule(rule_id="amps_stability", threshold=th, passed=actual <= th, actual_value=actual)


def _check_angle_consistency(features: Dict[str, Any], t: Optional[WeldTypeThresholds]) -> ScoreRule:
    actual = features.get("angle_max_deviation", 0.0)
    th = t.angle_warning_margin if t else ANGLE_CONSISTENCY_THRESHOLD
    return ScoreRule(rule_id="angle_consistency", threshold=th, passed=actual <= th, actual_value=actual)


def _check_thermal_symmetry(features: Dict[str, Any], t: Optional[WeldTypeThresholds]) -> ScoreRule:
    actual = features.get("north_south_delta_avg", 0.0)
    th = t.thermal_symmetry_warning_celsius if t else THERMAL_SYMMETRY_THRESHOLD
    return ScoreRule(rule_id="thermal_symmetry", threshold=th, passed=actual <= th, actual_value=actual)


def _check_heat_diss_consistency(features: Dict[str, Any], t: Optional[WeldTypeThresholds]) -> ScoreRule:
    actual = features.get("heat_diss_stddev", 0.0)
    th = t.heat_diss_consistency if t else HEAT_DISS_CONSISTENCY_THRESHOLD
```

```tsx
interface AngleArcDiagramProps {

export default function AngleArcDiagram({ angleTargetDees, className = '' }: AngleArcDiagramProps) {

  const cy = 30;
  const startAngle = 0;

  const x1 = cx + r * Math.cos(startAngle);

      <path d={d} fill="none" stroke="currentColor" strokeWidth="2" />
      <text x="30" y="38" textAnchor="middle" fontSize="10">{angleTargetDegrees}°</text>
```

```tsx
function isCompleteForm(f: Partial<WeldTypeThresholds>): f is WeldTypeThresholds {
  const isNum = (v: unknown) => typeof v === 'number' && Number.isFinite(v);

  const [fetchError, setFetchError] = useState<string | null>(null);
  const [active, setActive] = useState<typeof TABS[number]>('mig');
```

## Known Issues & Limitations

- **Multi-worker cache**: Single-process cache invalidates only in the worker that handled the PUT. Other workers retain stale thresholds until restart. For MVP: document; use single worker or add Redis to backlog.
- **Migration compatibility**: Assumes vanilla PostgreSQL. DBs with extensions, read replicas, or custom triggers may need migration adjustments.
- **Large-table migration**: Step 1.2 uses single `UPDATE sessions SET process_type = 'mig'`. On DBs with 100k+ sessions, this may lock the table and cause migration timeout or connection pool exhaustion. For large staging/prod: consider batched UPDATE (e.g. 5k rows per batch) or run during maintenance window.
- **AngleArcDiagram**: Validation limits angle_target to ≤90°; arc path assumes ≤90°.
- **SQLAlchemy 2.0**: Step 1.5 uses `db.query(WeldThresholdModel)`. Consider migrating to `db.execute(select(...))` style when updating SQLAlchemy.
