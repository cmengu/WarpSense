Organize issues by type:

**Type B: Insufficient Error Handling**

**Refined (with error handling):**
```javascript
function renderHeatmap() {
  // Guard: Check if data exists
  if (!heatmapData) {
    console.warn('⚠️ Heatmap data not loaded yet');
    return; // Exit gracefully
  }

  // Guard: Check if array is empty
  if (heatmapData.length === 0) {
    console.warn('⚠️ Heatmap data is empty');
    // Show "No data" UI instead of blank screen
    showNoDataMessage();
    return;
  }

  // Process data with error handling
  try {
    heatmapData.forEach(point => {
      // Validate each point
      if (!point || typeof point.temperature !== 'number') {
        console.error('Invalid point data:', point);
        return; // Skip this point
      }
      
      // Render point
      renderPoint(point);
    });
  } catch (error) {
    console.error('Error rendering heatmap:', error);
    // Fallback: show error state
    showErrorMessage('Failed to render heatmap');
  }
}
```

export const HeatmapShader = {
  uniforms: {
    temperatureMap: { value: null },
    minTemp: { value: 0.0 },
    maxTemp: { value: 500.0 }
  },
  vertexShader: `
    varying vec2 vUv;
    void main() {
      vUv = uv;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
    }
  `,
  fragmentShader: `
    uniform sampler2D temperatureMap;
    uniform float minTemp;
    uniform float maxTemp;
    varying vec2 vUv;
    
    vec3 temperatureToColor(float temp) {
      // Normalize temp to 0-1 range
      float t = clamp((temp - minTemp) / (maxTemp - minTemp), 0.0, 1.0);
      
      // Color gradient: blue → cyan → green → yellow → red → white
      if (t < 0.2) {
        return mix(vec3(0.0, 0.0, 0.545), vec3(0.0, 1.0, 1.0), t * 5.0);
      } else if (t < 0.4) {
        return mix(vec3(0.0, 1.0, 1.0), vec3(0.0, 1.0, 0.0), (t - 0.2) * 5.0);
      } else if (t < 0.6) {
        return mix(vec3(0.0, 1.0, 0.0), vec3(1.0, 1.0, 0.0), (t - 0.4) * 5.0);
      } else if (t < 0.8) {
        return mix(vec3(1.0, 1.0, 0.0), vec3(1.0, 0.0, 0.0), (t - 0.6) * 5.0);
      } else {
        return mix(vec3(1.0, 0.0, 0.0), vec3(1.0, 1.0, 1.0), (t - 0.8) * 5.0);
      }
    }
    
    void main() {
      float temp = texture2D(temperatureMap, vUv).r;
      vec3 color = temperatureToColor(temp);
      gl_FragColor = vec4(color, 1.0);
    }
  `
};

```typescript
function renderHeatmapFrame(frameIndex: number) {
  // Edge Case 1: Invalid frame index
  if (frameIndex < 0 || frameIndex >= totalFrames) {
    console.warn(`⚠️ Frame ${frameIndex} out of range (0-${totalFrames-1})`);
    return;
  }

  // Edge Case 2: No temperature data for this replay
  const temperatureData = replayStore.getTemperatureData();
  if (!temperatureData || temperatureData.length === 0) {
    console.warn('⚠️ No temperature data available');
    // Show default/neutral color
    setUniformTemperature(metalMesh, 70); // Room temperature
    return;
  }

  // Edge Case 3: Frame exists but has no data points
  const frameData = temperatureData[frameIndex];
  if (!frameData || frameData.points.length === 0) {
    console.warn(`⚠️ Frame ${frameIndex} has no data points`);
    // Interpolate from adjacent frames
    const interpolated = interpolateFromAdjacentFrames(frameIndex);
    applyTemperatureData(interpolated);
    return;
  }

  // Edge Case 4: Sparse data (missing measurements at some points)
  if (frameData.points.length < metalMesh.geometry.vertices.length * 0.5) {
    console.warn(`⚠️ Frame ${frameIndex} has sparse data, interpolating...`);
    const filled = fillSparsedData(frameData);
    applyTemperatureData(filled);
    return;
  }

  // Normal case: Full data available
  applyTemperatureData(frameData);
}

// Helper: Interpolate when frame has no data
function interpolateFromAdjacentFrames(frameIndex: number) {
  const prevFrame = findPreviousValidFrame(frameIndex);
  const nextFrame = findNextValidFrame(frameIndex);
  
  if (prevFrame && nextFrame) {
    // Linear interpolation between frames
    const t = (frameIndex - prevFrame.index) / (nextFrame.index - prevFrame.index);
    return interpolateFrames(prevFrame.data, nextFrame.data, t);
  } else if (prevFrame) {
    // Use previous frame
    return prevFrame.data;
  } else if (nextFrame) {
    // Use next frame
    return nextFrame.data;
  } else {
    // No data at all - use defaults
    return getDefaultTemperatureData();
  }
}
```

**Edge Cases Covered:**
1. ✅ Invalid frame index (negative or beyond range)
2. ✅ No temperature data loaded
3. ✅ Frame with zero data points
4. ✅ Sparse/incomplete data points
5. ✅ First frame (no previous frame to interpolate from)
6. ✅ Last frame (no next frame to interpolate from)
