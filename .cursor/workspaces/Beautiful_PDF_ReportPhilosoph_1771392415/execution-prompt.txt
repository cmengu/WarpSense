import { Session, Frame, ThermalSnapshot } from '@/types/session';

/**
 * Generate expert session with high-quality welding data
 * Score: 94/100
 * Duration: 15 seconds
 * Frame rate: 30 FPS
 */
export function generateExpertSession(): Session {
  const frames: Frame[] = [];
  const sessionId = 'demo-expert';
  const startTime = Date.now();
  
  // Generate 450 frames (15 seconds at 30 FPS)
  for (let i = 0; i < 450; i++) {
    const timestamp = startTime + (i * 33.33); // ~30 FPS
    
    frames.push({
      id: `${sessionId}-frame-${i}`,
      session_id: sessionId,
      timestamp,
      frame_number: i,
      
      // Expert maintains 45° angle with minimal deviation
      angle_degrees: 45 + (Math.random() - 0.5) * 2, // ±1°
      
      // Consistent travel speed
      travel_speed: 12.5 + (Math.random() - 0.5) * 0.5, // ~12.5 mm/s
      
      // Thermal data every 5 frames (simulating thermal camera rate)
      thermal_snapshot: (i % 5 === 0) ? generateThermalSnapshot(1200, 50) : null,
      
      // Torch position progresses steadily
      torch_position: {
        x: 100 + (i * 0.5),
        y: 200,
        z: 150
      }
    });
  }
  
  return {
    id: sessionId,
    welder_name: 'Expert Welder',
    start_time: startTime,
    end_time: startTime + 15000,
    frames,
    score: 94,
    metadata: {
      quality_grade: 'A',
      technique: 'GMAW',
      material: 'Steel'
    }
  };
}

/**
 * Generate novice session with erratic welding patterns
 * Score: 42/100
 */
export function generateNoviceSession(): Session {
  const frames: Frame[] = [];
  const sessionId = 'demo-novice';
  const startTime = Date.now();
  
  for (let i = 0; i < 450; i++) {
    const timestamp = startTime + (i * 33.33);

  }

/**
 * Generate thermal snapshot with realistic temperature distribution
 */
function generateThermalSnapshot(
  centerTemp: number,
  variation: number
): ThermalSnapshot {
  const data: number[][] = [];
  
  // 48x48 grid
  for (let y = 0; y < 48; y++) {
    const row: number[] = [];
    for (let x = 0; x < 48; x++) {
      // Distance from center
      const dx = x - 24;
      const dy = y - 24;
      const distance = Math.sqrt(dx * dx + dy * dy);
      
      // Temperature decreases with distance from center
      const temp = centerTemp - (distance * 10) + (Math.random() - 0.5) * variation;
      row.push(Math.max(400, Math.min(1600, temp))); // Clamp to realistic range
    }
    data.push(row);
  }
  
  return {
    data,
    width: 48,
    height: 48,
    min_temp: 400,
    max_temp: 1600,
    center_temp: centerTemp
  };
}

Step 3: Chart → PNG (the tricky bit)
@react-pdf/renderer can't render Recharts. You need to pre-render the trend chart to a base64 PNG. Two options:
Option A (fast, good enough): Use html-to-image on the client to capture the existing LineChart div into a PNG, pass it as base64 to the PDF route. Ship this first.
Option B (clean, proper): Build a <canvas> chart in the API route using chartjs-node-canvas (server-side Chart.js). More work, no DOM dependcy.
Start with Option A.

typescript// app/api/welder-report-pdf/route.ts
import { renderToBuffer } from '@react-pdf/renderer';
import { WelderReportPDF } from '@/components/pdf/WelderReportPDF';

export async function POST(req: Request) {
  const { welder, score, feedback, chartDataUrl } = await req.json();
  
  const buffer = await renderToBuffer(
    <WelderReportPDF welder={welder} score={score} 
                     feedback={feedback} chartDataUrl={chartDataUrl} />
  );
  
  return new Response(buffer, {
    headers: {
      'Content-Type': 'application/pdf',
      'Content-Disposition': `attachment; filename="${welder.name}-report.pdf"`,
    },
  });
}
